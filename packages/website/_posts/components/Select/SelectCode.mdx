### react-hook-form example

```jsx live reactHookForm
<>
  <Select
    labelText="The uncontrolled Input"
    helperText="Optional helperText"
    {...register("inputname")}
  >
    <SelectItem text="Option 1" value="1" />
    <SelectItem text="Option 2" value="2" />
    <SelectItem text="Option 3" value="3" />
  </Select>

  <Controller
    name="inputnameB"
    control={control}
    render={({ field }) => {
      return (
        <Select
          labelText="The controlled Input"
          helperText="Optional helperText"
          {...field}
          onChange={(e) => field.onChange(e.target.value)}
          value={field.value}
        >
          <SelectItem text="Option 1" value="1" />
          <SelectItem text="Option 2" value="2" />
          <SelectItem text="Option 3" value="3" />
        </Select>
      );
    }}
  />
</>
```
